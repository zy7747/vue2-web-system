```html
  <video loop="loop" class="video" ref="video" width="100%">
      <source src="/village-low.mp4" type="video/mp4" />
    </video>
<canvas ref="canvas" class="canvas"></canvas> 
```



```js
  data() {
    return {
      video: null, // 用于存储视频元素的引用
      canvas: null,
      ctx: null,
    };
  },  

// 在 Canvas 上绘制视频帧
    draw() {
      this.ctx.drawImage(
        this.video,
        0,
        0,
        this.canvas.width,
        this.canvas.height
      );
      requestAnimationFrame(this.draw); // 循环绘制，实现视频的连续播放效果
    },
    //播放视频
    play() {
      this.canvas = this.$refs.canvas;
      this.ctx = this.canvas.getContext("2d");

      const dpr = window.devicePixelRatio || 1; // 获取设备像素比
      const width = this.canvas.clientWidth * dpr; // 根据 Canvas 元素在页面上的实际尺寸计算宽度
      const height = this.canvas.clientHeight * dpr; // 根据 Canvas 元素在页面上的实际尺寸计算高度

      this.canvas.width = width; // 设置 Canvas 的宽度为实际宽度 * 设备像素比
      this.canvas.height = height; // 设置 Canvas 的高度为实际高度 * 设备像素比

      this.ctx.scale(dpr, dpr); // 缩放绘图上下文以适应设备像素比

      this.video = this.$refs.video;

      //视频播放
      this.video.play();
      //绘制图像
      this.draw();
    },
    handleCanPlay(event) {
      this.play();
    },
```

```css
.canvas {
  position: fixed;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
  object-fit: fill; /* 禁用 Canvas 的尺寸缩放 */
}

  width: 100%;
  height: 100%;
  object-fit: cover;
  opacity: 0;
}
```
